CC = gcc
CFLAGS = -Wall -Wextra -pedantic

all: marvelmind_c parser main

marvelmind_c:
	gcc -g -pthread -c marvelmind.c -o marvelmind.o
	gcc -g -pthread -c example.c -o example.o
	gcc -o marvelmind_c example.o marvelmind.o -pthread

main: main.o
	$(CC) $(CFLAGS) $^ -o $@

main.o: main.c
	$(CC) $(CFLAGS) -c $< -o $@

parser: parser.o
	$(CC) $(CFLAGS) $^ -o $@

parser.o: parser.c
	$(CC) $(CFLAGS) -c $< -o $@

clean:
	rm -rf *.o
